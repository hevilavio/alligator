## Instalando a lib do dht11

git clone https://github.com/adafruit/Adafruit_Python_DHT.git
cd Adafruit_Python_DHT
sudo apt-get update && sudo apt-get install build-essential python-dev python-openssl
sudo python setup.py install

## Para testar o funcionamento do sensor
cd examples
sudo ./AdafruitDHT.py 11 4
sudo ./AdafruitDHT.py <tipo_do_sensor> <porta_gpio>


## Instalando bibliteca de mqtt
sudo ./install_dependencies.sh


## Instalando o programa no raspberry
